#+begin_quote

git clone --depth 1 https://github.com/raysan5/raylib.git raylib

cd raylib/src/

# ===== To make the static version.

# ===== Static version.

  int const aNumber2 = 99;


  // The way I memorize is constant pointer VS pointer to constant,
  // is that asterisk(*) serves to take emphasis away,
  // so you ignore the * in the declaration 

  int const *imInside = &aNumber2;

  imInside = &aNumber1;


  // this will fail
  // *imInside = 2000;






  int* const imOutside = &aNumber2;

  ,*imOutside = 2000;

  // this will fail
  // imOutside = &aNumber1;
